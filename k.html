import pandas as pd

import seaborn as sns
dataframe = pd.read_csv("Zomato data .csv")
print(dataframe.head())
def handleRate(value):
    value=str(value).split('/')
    value=value[0];
    return float(value)

dataframe['rate']=dataframe['rate'].apply(handleRate)
